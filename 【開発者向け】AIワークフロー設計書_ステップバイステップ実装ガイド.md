    """ãƒ‡ãƒ¼ã‚¿ãƒ™ãƒ¼ã‚¹æ¥ç¶šå–å¾—"""

            obs.field,
            str(obs.value),
            obs.unit,
            obs.source_tag.value,
            obs.evidence,
            obs.as_of.isoformat(),
            obs.disclosure.value,
            obs.confidence,
            obs.notes

        conn.commit()

                disclosure=DisclosureLevel.LP,
                confidence=obs_data.get("confidence", 0.9),
